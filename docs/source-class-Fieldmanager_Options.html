394:     /**
395:      * Helper to support recursive building of a hierarchical taxonomy list.
396:      * @param array $parent_terms
397:      * @param array $tax_args as used in top-level get_terms() call.
398:      * @param int $depth current recursive depth level.
399:      * @return array of terms or false if no children found.
400:      */
401:     protected function build_hierarchical_term_data( $parent_terms, $tax_args, $depth ) {
402:         
403:         // Walk through each term passed, add it (at current depth) to the data stack.
404:         foreach ( $parent_terms as $term ) {
405:             $taxonomy_data = get_taxonomy( $term->taxonomy );
406:             $prefix = '';
407:             
408:             // Prefix term based on depth. For $depth = 0, prefix will remain empty.
409:             for ( $i = 0; $i < $depth; $i++ ) {
410:                 $prefix .= '--';
411:             }
412:             
413:             $this->add_option_data( $prefix . ' ' . $term->name, $term->term_id, $taxonomy_data->label, $taxonomy_data->name );

416:             $tax_args['parent'] = $term->term_id;
417:             $child_terms = get_terms( $this->taxonomy, $tax_args );
418:             if ( $this->taxonomy_hierarchical_depth == 0 || $depth + 1 < $this->taxonomy_hierarchical_depth ) {

420:                     $this->build_hierarchical_term_data( $child_terms, $this->taxonomy_args, $depth + 1 );

422:             }
423:         }
